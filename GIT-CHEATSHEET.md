git config --global user.name "Имя" — установить имя
git config --global user.email "email@example.com" — установить email
git config --list — показать текущие настройки
git init — создать новый локальный репозиторий в текущей папке
git clone <url> — скопировать удаленный репозиторий на компьютер
git status — проверить состояние файлов (изменены, добавлены, готовы к коммиту)
git add <file> — добавить конкретный файл в индекс (Staging area)
git add . — добавить все измененные файлы.
